class Solution {
    // Function to rotate an array by d elements in counter-clockwise direction.
    static void rotateArr(int arr[], int d) {
        // add your code here
        int n=arr.length;
        int[] temp=new int[d];
        d=d%n;
        for(int i=0;i<d;i++){
            temp[i]=arr[i];
        }
        int in=0;
        for(int i=d;i<n;i++)
            arr[in++]=arr[i];
        for(int i=0;i<d%n;i++)
            arr[in++]=temp[i];
        
        
    }
}
